# paste in the complete backend - this has no ingress/external access, but is otherwise
# very similar to the frontend:

apiVersion: apps/v1
kind: Deployment
metadata:
  name: hip-local-api
  labels:
    app: hip-local-api
spec:
  replicas: 3
  selector:
    matchLabels:
      app: hip-local-api
  template:
    metadata:
      labels:
        app: hip-local-api
    spec:
      containers:
      - name: hip-local-api
        image: gcr.io/kr-dr-temp-hip/api:v0.1
        env:
        - name: GOOGLE_CLOUD_PROJECT
          value: "kr-dr-temp-hip"
        ports:
        - containerPort: 8081
          name: "backend-server"
        resources:
          requests:
            memory: "256Mi"
            cpu: "0.1"
          limits:
            memory: "512Mi"
            cpu: "0.5"
        readinessProbe:
          httpGet:
            path: /
            port: 8081
          initialDelaySeconds: 30
          periodSeconds: 30
        livenessProbe:
          httpGet:
            path: /
            port: 8081
          initialDelaySeconds: 15
          periodSeconds: 15
---

apiVersion: v1
kind: Service
metadata: 
  name: hip-local-api-svc
spec: 
  ports: 
     -  port: 8081
        protocol: TCP
        targetPort: 8081
  selector: 
    app: hip-local-api
  type: NodePort


# deploy backend with apply (note, we can use apply to create...) 

kubectl apply -f kubernetes-config.yaml
