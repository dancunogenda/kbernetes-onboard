# We will add liveness and readiness probes pointing at the /health endpoint. 
# This endpoint returns an empty array (and 200 status) if the app is running

# Navigate to the frontend yaml again and add the following readiness probe at the bottom of 
# the  deployment, immediately above the ---.  It is at the same indent level as resources 

        readinessProbe:
          httpGet:
            path: /health
            port: 8080
          initialDelaySeconds: 30
          periodSeconds: 30

# We also want a liveness probe.  Add it immediately after the readiness probe

        livenessProbe:
          httpGet:
            path: /health
            port: 8080
          initialDelaySeconds: 15
          periodSeconds: 15

# Save the file and apply the new configuration.

kubectl apply -f kubernetes-config.yaml

# Paste in the complete backend. It has no ingress/external access, 
# but is otherwise very similar to the frontend:

apiVersion: apps/v1
kind: Deployment
metadata:
  name: hip-local-api
  labels:
    app: hip-local-api
spec:
  replicas: 3
  selector:
    matchLabels:
      app: hip-local-api
  template:
    metadata:
      labels:
        app: hip-local-api
    spec:
      containers:
      - name: hip-local-api
        image: gcr.io/kr-dr-temp-hip/api:v0.1
        env:
        - name: GOOGLE_CLOUD_PROJECT
          value: "kr-dr-temp-hip"
        ports:
        - containerPort: 8081
          name: "backend-server"
        resources:
          requests:
            memory: "256Mi"
            cpu: "0.1"
          limits:
            memory: "512Mi"
            cpu: "0.5"
        readinessProbe:
          httpGet:
            path: /
            port: 8081
          initialDelaySeconds: 15
          periodSeconds: 15
        livenessProbe:
          httpGet:
            path: /
            port: 8081
          initialDelaySeconds: 15
          periodSeconds: 15
---

apiVersion: v1
kind: Service
metadata: 
  name: hip-local-api-svc
spec: 
  ports: 
     -  port: 8081
        protocol: TCP
        targetPort: 8081
  selector: 
    app: hip-local-api
  type: NodePort
